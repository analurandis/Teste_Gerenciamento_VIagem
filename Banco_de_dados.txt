CREATE TABLE TGV_MOTORISTA 
(
  CODIGO INT  GENERATED BY DEFAULT ON NULL AS IDENTITY
, NOME VARCHAR2(120) NOT NULL 
, RUA VARCHAR2(120) 
, NUMERO VARCHAR2(10) 
, BAIRRO VARCHAR2(50) 
, CIDADE VARCHAR2(50) 
, ESTADO VARCHAR2(2) 
, CEP VARCHAR2(14) 
, CONSTRAINT TGV_MOTORISTA_PK PRIMARY KEY 
  (
    CODIGO 
  )
  ENABLE 
);


CREATE TABLE TGV_CAMINHAO 
(
  CODIGO INT  GENERATED BY DEFAULT ON NULL AS IDENTITY
, MARCA VARCHAR2(60) NOT NULL 
, MODELO VARCHAR2(60) 
, PLACA VARCHAR2(10) 
, EIXO INT
, PESO  FLOAT(5)
, CONSTRAINT TGV_CAMINHAO_PK PRIMARY KEY 
  (
    CODIGO 
  )
  ENABLE 
);

CREATE TABLE TGV_VIAGEM
(
  CODIGO INT  GENERATED BY DEFAULT ON NULL AS IDENTITY
, DATAHORA DATE 
, LOCAL_ENTREGA VARCHAR2(120) 
, LOCAL_SAIDA VARCHAR2(120) 
, KM_PERCURSO FLOAT
, MOTORISTA_CODIGO INT
, CONSTRAINT TGV_VIAGEM_PK PRIMARY KEY 
  (
    CODIGO 
  ),
  CONSTRAINT TGV_VIAGEM_MOTORISTA_FK
   FOREIGN KEY (MOTORISTA_CODIGO)
   REFERENCES TGV_MOTORISTA (CODIGO)
);
